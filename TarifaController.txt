package controller;

import java.util.List;
import model.Tarifa;

/**
 *
 * @author jpescola
 */
public class TarifaController extends Controller<Tarifa> {

    @Override
    public boolean salvar(Tarifa t) {
        return super.salvar(t);
    }

    @Override
    public boolean excluir(Tarifa t) {
        return super.excluir(t);
    }

    public List<Tarifa> buscar(String campo, String valor, String tipo) {
        return super.listar(Tarifa.class, campo, valor, tipo);
    }

    public Tarifa get(int id) {
        return super.get(Tarifa.class, id);
    }

    public List<Tarifa> listar() {
        return super.listar(Tarifa.class, "dataCriacao", "", "");
    }
}
