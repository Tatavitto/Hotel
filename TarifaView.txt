    private void loadComboClientes() {
        jComboHospede.removeAllItems(); // liimpa o combo
        jComboHospede.addItem("");
        // atualiza o arraylist de quartos com os dados do bd
        clientes = new ClienteController().listar();
        if (!clientes.isEmpty()) { // caso existam quartos cadastrados
            // adiciona os nomes dos quartos no combo
            for (Cliente f : clientes) { // percorre o arraylist de quartos
                jComboHospede.addItem(f.getNome()); // adiciona o nome do quarto ao combo
            }
        }
    }

    private void loadComboQuartos() {
        jComboQuarto.removeAllItems(); // liimpa o combo
        jComboQuarto.addItem("");
        // atualiza o arraylist de quartos com os dados do bd
        quartos = new QuartoController().listar();
        if (!quartos.isEmpty()) { // caso existam quartos cadastrados
            // adiciona os nomes dos quartos no combo
            for (Quarto f : quartos) { // percorre o arraylist de quartos
                jComboQuarto.addItem(f.getNome()); // adiciona o nome do quarto ao combo
            }
        }
    }

    jComboDFIXO.setEnabled(false);
    }

    private void salvar() {
        if (cliente == null || quarto == null) {
            return;  
        }
        
        try {
            tarifa.setCliente(cliente); // define o cliente da reserva
            tarifa.setQuarto(quarto); // define o quarto da reserva
            tarifa.setDescontoP(isDescontoPersonalizado);
            tarifa.setDescontoF(isDescontoFixo);
            
            
        double valorDesconto = 0.0;

        if (isDescontoPersonalizado) {
            String texto = jTextField1.getText().replace(",", ".");
            if (!texto.isEmpty()) {
                valorDesconto = Double.parseDouble(texto);
            }
        } else if (isDescontoFixo) {
            String selecionado = (String) jComboDFIXO.getSelectedItem();
            if (selecionado != null && !selecionado.isEmpty()) {
                selecionado = selecionado.replace("%", "").trim();
                valorDesconto = Double.parseDouble(selecionado) / 100.0;
            }
        }

        tarifa.setValorDesconto((float) valorDesconto);
        
        
        } catch (Exception e) {
            e.printStackTrace();
        }
        
        
    }
